Login Tests

Basic Login Test

Basic login functionality with valid credentials 
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager
import time

def test_basic_login():
    # Setup WebDriver
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    
    try:
        # Navigate to login page
        driver.get("http://localhost:8000/login-scenarios.html")
        
        # Wait for page to load
        wait = WebDriverWait(driver, 10)
        
        # Find and fill username field
        username_field = wait.until(EC.presence_of_element_located((By.ID, "username")))
        username_field.clear()
        username_field.send_keys("testuser")
        
        # Find and fill password field
        password_field = driver.find_element(By.ID, "password")
        password_field.clear()
        password_field.send_keys("password123")
        
        # Click login button
        login_button = driver.find_element(By.ID, "login-btn")
        login_button.click()
        
        # Wait for success message
        success_message = wait.until(EC.presence_of_element_located((By.CLASS_NAME, "success-message")))
        
        # Verify login was successful
        assert "Login successful" in success_message.text
        print("✅ Basic login test passed!")
        
    except Exception as e:
        print(f"❌ Login test failed: {e}")
        
    finally:
        driver.quit()

if __name__ == "__main__":
    test_basic_login()

def test_registration_form():
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    
    try:
        driver.get("http://localhost:8000/form-handling.html")
        wait = WebDriverWait(driver, 10)
        
        # Fill registration form
        first_name = wait.until(EC.presence_of_element_located((By.ID, "first-name")))
        first_name.clear()
        first_name.send_keys("John")
        
        last_name = driver.find_element(By.ID, "last-name")
        last_name.clear()
        last_name.send_keys("Doe")
        
        email = driver.find_element(By.ID, "email")
        email.clear()
        email.send_keys("john.doe@example.com")
        
        phone = driver.find_element(By.ID, "phone")
        phone.clear()
        phone.send_keys("1234567890")
        
        # Select gender
        gender_male = driver.find_element(By.ID, "gender-male")
        gender_male.click()
        
        # Select country from dropdown
        country_select = Select(driver.find_element(By.ID, "country"))
        country_select.select_by_visible_text("United States")
        
        # Check terms checkbox
        terms_checkbox = driver.find_element(By.ID, "terms")
        if not terms_checkbox.is_selected():
            terms_checkbox.click()
        
        # Submit form
        submit_button = driver.find_element(By.ID, "submit-btn")
        submit_button.click()
        
        # Wait for success message
        success_message = wait.until(EC.presence_of_element_located((By.CLASS_NAME, "success-message")))
        assert "Registration successful" in success_message.text
        print("✅ Registration form test passed!")
        
    except Exception as e:
        print(f"❌ Registration form test failed: {e}")
        
    finally:
        driver.quit()

from selenium.webdriver.common.alert import Alert

Dynamic Tables

Table Data Extraction Test

Extract and verify data from dynamic tables 
def test_table_data_extraction():
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    
    try:
        driver.get("http://localhost:8000/dynamic-tables.html")
        wait = WebDriverWait(driver, 10)
        
        # Wait for table to load
        table = wait.until(EC.presence_of_element_located((By.ID, "data-table")))
        
        # Get all rows
        rows = table.find_elements(By.TAG_NAME, "tr")
        
        # Skip header row
        data_rows = rows[1:]
        
        # Extract data from first row
        first_row_cells = data_rows[0].find_elements(By.TAG_NAME, "td")
        name = first_row_cells[0].text
        email = first_row_cells[1].text
        age = first_row_cells[2].text
        
        print(f"First row data: {name}, {email}, {age}")
        
        # Verify table has data
        assert len(data_rows) > 0, "Table should have data rows"
        assert name != "", "Name should not be empty"
        
        # Test sorting functionality
        name_header = driver.find_element(By.ID, "sort-name")
        name_header.click()
        
        # Wait for sorting to complete
        time.sleep(1)
        
        # Verify sorting worked
        sorted_rows = table.find_elements(By.TAG_NAME, "tr")[1:]
        first_sorted_name = sorted_rows[0].find_elements(By.TAG_NAME, "td")[0].text
        print(f"First name after sorting: {first_sorted_name}")
        
        print("✅ Table data extraction test passed!")
        
    except Exception as e:
        print(f"❌ Table data extraction test failed: {e}")
        
    finally:
        driver.quit()

Multiple File Upload Test

Upload multiple files and verify progress 
def test_multiple_file_upload():
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    
    try:
        driver.get("http://localhost:8000/file-upload.html")
        wait = WebDriverWait(driver, 10)
        
        # Create multiple test files
        test_files = []
        for i in range(3):
            file_path = os.path.join(os.getcwd(), f"test_file_{i}.txt")
            with open(file_path, "w") as f:
                f.write(f"This is test file {i}")
            test_files.append(file_path)
        
        # Find multiple file input
        file_input = wait.until(EC.presence_of_element_located((By.ID, "multiple-files")))
        
        # Upload multiple files
        file_input.send_keys("\n".join(test_files))
        
        # Wait for all uploads to complete
        progress_bars = driver.find_elements(By.CLASS_NAME, "progress-bar")
        assert len(progress_bars) == 3, "Should have 3 progress bars"
        
        # Wait for completion
        time.sleep(3)
        
        # Verify all files uploaded
        success_messages = driver.find_elements(By.CLASS_NAME, "upload-success")
        assert len(success_messages) == 3, "All 3 files should be uploaded"
        
        # Clean up test files
        for file_path in test_files:
            os.remove(file_path)
        print("✅ Multiple file upload test passed!")
        
    except Exception as e:
        print(f"❌ Multiple file upload test failed: {e}")
        
    finally:
        driver.quit()

from selenium.webdriver.common.action_chains import ActionChains

def test_drag_drop_upload():
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    
    try:
        driver.get("http://localhost:8000/drag-drop.html")
        wait = WebDriverWait(driver, 10)
        
        # Create test file
        test_file_path = os.path.join(os.getcwd(), "drag_drop_test.txt")
        with open(test_file_path, "w") as f:
            f.write("Test file for drag and drop")
        
        # Find drop zone
        drop_zone = wait.until(EC.presence_of_element_located((By.ID, "single-file-dropzone")))
        
        # Create ActionChains for drag and drop
        actions = ActionChains(driver)
        
        # Simulate drag and drop (this is a simplified version)
        # In real scenarios, you might need to use JavaScript or other methods
        actions.drag_and_drop_by_offset(drop_zone, 0, 0).perform()
        
        # Alternative: Use JavaScript to trigger file drop
        driver.execute_script("""
            var dropZone = arguments[0];
            var fileInput = document.createElement('input');
            fileInput.type = 'file';
            fileInput.files = arguments[1];
            dropZone.appendChild(fileInput);
            var event = new Event('change', { bubbles: true });
            fileInput.dispatchEvent(event);
        """, drop_zone, test_file_path)
        
        # Wait for upload success
        upload_success = wait.until(EC.presence_of_element_located((By.CLASS_NAME, "upload-success")))
        assert "File uploaded successfully" in upload_success.text
        
        # Clean up
        os.remove(test_file_path)
        print("✅ Drag & drop upload test passed!")
        
    except Exception as e:
        print(f"❌ Drag & drop upload test failed: {e}")
        
    finally:
        driver.quit()

from datetime import datetime, timedelta

def test_date_input():

        date_with_limits.send_keys(valid_date)

import pytest
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from webdriver_manager.chrome import ChromeDriverManager

@pytest.fixture(scope="function")
def driver():
    """Setup WebDriver for each test"""
    service = Service(ChromeDriverManager().install())
    driver = webdriver.Chrome(service=service)
    driver.maximize_window()
    yield driver
    driver.quit()

@pytest.fixture(scope="session")
def base_url():
    """Base URL for the application"""
    return "http://localhost:8000"
