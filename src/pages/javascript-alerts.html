Selenium Code Examples

Basic Alert Test

// Basic Alert Test
@Test
public void testBasicAlert() {
    driver.findElement(By.id("basic-alert-btn")).click();
    
    // Switch to alert
    Alert alert = driver.switchTo().alert();
    
    // Get alert text
    String alertText = alert.getText();
    assertEquals("This is a basic alert!", alertText);
    
    // Accept alert
    alert.accept();
}

Confirm Dialog Test

// Confirm Dialog Test
@Test
public void testConfirmDialog() {
    driver.findElement(By.id("confirm-alert-btn")).click();
    
    Alert alert = driver.switchTo().alert();
    String alertText = alert.getText();
    assertTrue(alertText.contains("Are you sure"));
    
    // Accept (OK)
    alert.accept();
    
    // Or dismiss (Cancel)
    // alert.dismiss();
}

Prompt Dialog Test

// Prompt Dialog Test
@Test
public void testPromptDialog() {
    driver.findElement(By.id("prompt-alert-btn")).click();
    
    Alert alert = driver.switchTo().alert();
    
    // Send text to prompt
    alert.sendKeys("Test User Input");
    
    // Accept prompt
    alert.accept();
    
    // Verify result
    String result = driver.findElement(By.id("prompt-result")).getText();
    assertEquals("You entered: Test User Input", result);
}

Alert Handling Test

// Alert Handling Test
@Test
public void testAlertHandling() {
    // Check if alert is present
    try {
        Alert alert = driver.switchTo().alert();
        String text = alert.getText();
        alert.accept();
        System.out.println("Alert text: " + text);
    } catch (NoAlertPresentException e) {
        System.out.println("No alert present");
    }
}
