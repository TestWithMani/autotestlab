Selenium Code Examples

Valid Login Test

// Valid Login Test
@Test
public void testValidLogin() {
    driver.findElement(By.id("username"))
          .sendKeys("admin");
    driver.findElement(By.id("password"))
          .sendKeys("password123");
    driver.findElement(By.id("login-button"))
          .click();
    
    // Verify success message
    WebElement message = driver.findElement(By.id("login-message"));
    assertTrue(message.isDisplayed());
    assertTrue(message.getText().contains("Login successful"));
}

Invalid Login Test

// Invalid Login Test
@Test
public void testInvalidLogin() {
    driver.findElement(By.id("username"))
          .sendKeys("wrong");
    driver.findElement(By.id("password"))
          .sendKeys("wrong");
    driver.findElement(By.id("login-button"))
          .click();
    
    // Verify error message
    WebElement message = driver.findElement(By.id("login-message"));
    assertTrue(message.isDisplayed());
    assertTrue(message.getText().contains("Invalid credentials"));
}

Empty Fields Test

// Empty Fields Test
@Test
public void testEmptyFields() {
    // Leave fields empty
    driver.findElement(By.id("login-button"))
          .click();
    
    // Verify validation messages
    WebElement usernameError = driver.findElement(By.id("username-error"));
    WebElement passwordError = driver.findElement(By.id("password-error"));
    
    assertTrue(usernameError.isDisplayed());
    assertTrue(passwordError.isDisplayed());
}

Password Toggle Test

// Password Toggle Test
@Test
public void testPasswordToggle() {
    WebElement passwordField = driver.findElement(By.id("password"));
    WebElement toggleButton = driver.findElement(By.id("toggle-password"));
    
    // Initially password type
    assertEquals("password", passwordField.getAttribute("type"));
    
    // Click toggle button
    toggleButton.click();
    
    // Now should be text type
    assertEquals("text", passwordField.getAttribute("type"));
}
